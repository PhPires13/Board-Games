103 TEST_CASE("Get game state") {
104 constexpr int height = 7, width = 5;
105 BoardGame boardGame(Player("Nick1", "Name1"), Player("Nick2", "Name2"), height, width);
106 for (uint32_t i =0; i < height; i++) {
107 for (uint32_t j = 0; j < width; j++) {
108 boardGame.makeMove({j}, 'X');
109 if (i != height-1 || j != width-1) CHECK(boardGame.getGameState({}) == GameState::NOT_OVER);
110 }
